Visually join an input with adjacent buttons — borders merge seamlessly for a single cohesive control.

## Installation

### Barrel

```tsx
import { InputGroup } from "@catiarodrigues/goldfinch";
```

Or as a compound sub-component:

```tsx
import { Input } from "@catiarodrigues/goldfinch";
// Input.Group is available as a sub-component
```

### Granular

```tsx
import { InputGroup } from "@catiarodrigues/goldfinch/components/input-group";
```

## Usage

```tsx
import { InputGroup, Input, Button } from "@catiarodrigues/goldfinch";

export default function Example() {
  return (
    <InputGroup>
      <Button variant="secondary" onClick={() => alert("Search")}>Search</Button>
      <Input placeholder="Search..." aria-label="Search" />
    </InputGroup>
  );
}
```

</Callout>
