Display a read-only value alongside a copy button — perfect for API tokens, install commands, and shareable IDs.

## Installation

### Barrel

```tsx
import { CopyToClipboard } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { CopyToClipboard } from "@catiarodrigues/goldfinch/components/copy-to-clipboard";
```

## Usage

```tsx
import { CopyToClipboard } from "@catiarodrigues/goldfinch";

export default function Example() {
  return <CopyToClipboard text="npm install @catiarodrigues/goldfinch" />;
}
```

## Examples

<LiveCopyToClipboard title="Copy to Clipboard" />

