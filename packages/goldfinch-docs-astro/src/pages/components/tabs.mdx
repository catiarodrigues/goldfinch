Organize related content into switchable panels. Two visual modes: `segmented` (pill-style grouped tabs) and `underline` (classic tab bar).

## Installation

### Barrel

```tsx
import { Tabs } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { Tabs } from "@catiarodrigues/goldfinch/components/tabs";
```

## Usage

```tsx
import { Tabs } from "@catiarodrigues/goldfinch";
import { useState } from "react";

export default function Example() {
  return (
    <Tabs
      tabs={[
        { value: "overview", label: "Overview" },
        { value: "details", label: "Details" },
      ]}
      defaultValue="overview"
      variant="segmented"
    />
  );
}
```

## Examples

<LiveTabs title="Tabs" />

Define tabs via the `tabs` prop array. Use `value` and `onValueChange` for controlled mode, or `defaultValue` for uncontrolled.

```tsx
<Tabs
  tabs={[
    { value: "tab-1", label: "Tab 1" },
    { value: "tab-2", label: "Tab 2" },
  ]}
  variant="underline"
/>
```

<Callout type="info">
  The `segmented` variant renders tabs as a segmented control, while `underline` uses underlined tab labels.
</Callout>
