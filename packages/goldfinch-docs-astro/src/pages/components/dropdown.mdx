Present a list of actions or links in a positioned overlay that opens on click.

## Installation

### Barrel

```tsx
import { DropdownMenu } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { DropdownMenu } from "@catiarodrigues/goldfinch/components/dropdown";
```

## Usage

```tsx
import { DropdownMenu, Button } from "@catiarodrigues/goldfinch";

export default function Example() {
  return (
    <DropdownMenu>
      <DropdownMenu.Trigger>
        <Button variant="ghost">Actions</Button>
      </DropdownMenu.Trigger>
      <DropdownMenu.Portal>
        <DropdownMenu.Content align="start">
          <DropdownMenu.Item onClick={() => {}}>Edit</DropdownMenu.Item>
          <DropdownMenu.Item onClick={() => {}}>Duplicate</DropdownMenu.Item>
          <DropdownMenu.Separator />
          <DropdownMenu.Item onClick={() => {}} variant="danger">Delete</DropdownMenu.Item>
        </DropdownMenu.Content>
      </DropdownMenu.Portal>
    </DropdownMenu>
  );
}
```
