Hide secondary content behind a trigger — ideal for settings panels, FAQ items, or long metadata sections.

## Installation

### Barrel

```tsx
import { Collapsible } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { Collapsible } from "@catiarodrigues/goldfinch/components/collapsible";
```

## Usage

```tsx
import { Collapsible, Button } from "@catiarodrigues/goldfinch";

export default function Example() {
  return (
    <Collapsible>
      <Collapsible.Trigger render={<Button variant="ghost" />}>
        Toggle
      </Collapsible.Trigger>
      <Collapsible.Panel>
        <div>Hidden content revealed.</div>
      </Collapsible.Panel>
    </Collapsible>
  );
}
```
