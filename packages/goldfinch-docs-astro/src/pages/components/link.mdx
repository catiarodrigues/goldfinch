Style anchor elements with Goldfinch's link tokens for consistent color, underline, and hover behavior.

## Installation

### Barrel

```tsx
import { Link } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { Link } from "@catiarodrigues/goldfinch/components/link";
```

## Usage

```tsx
import { Link } from "@catiarodrigues/goldfinch";

export default function Example() {
  return (
    <Link href="https://example.com" target="_blank" rel="noopener noreferrer">
      Visit Example
    </Link>
  );
}
```
