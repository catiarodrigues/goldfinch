A horizontal top bar for app-level navigation — brand logo, nav links, and right-aligned actions.

## Installation

### Barrel

```tsx
import { TopNavigation } from "@catiarodrigues/goldfinch";
```

### Granular

```tsx
import { TopNavigation } from "@catiarodrigues/goldfinch/components/top-navigation";
```

## Usage

```tsx
import { TopNavigation, Button } from "@catiarodrigues/goldfinch";

export default function AppBar() {
  return (
    <TopNavigation>
      <TopNavigation.Logo>
        <span className="text-lg font-bold">MyApp</span>
      </TopNavigation.Logo>
      <TopNavigation.Nav>
        <TopNavigation.NavItem active>Dashboard</TopNavigation.NavItem>
        <TopNavigation.NavItem>Settings</TopNavigation.NavItem>
      </TopNavigation.Nav>
      <TopNavigation.Actions>
        <Button variant="ghost" size="sm">Log in</Button>
      </TopNavigation.Actions>
    </TopNavigation>
  );
}
```
